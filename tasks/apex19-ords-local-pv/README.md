# Deploy Oracle Apex 19.1 and ORDS webapp on Tomcat in 2 containers, database files are stored on a local directory that is defined as a persistent volume in Kubernetes

- create a local directory /newdisk on a worker node for pv local-apex19-db
```
sudo mkdir /newdisk
sudo chmod 777 /newdisk
```
- create storage class, persistent volume and persistent volume claim
```
cat<<EOF | kubectl apply -f -
kind: StorageClass
apiVersion: storage.k8s.io/v1
metadata:
  name: local
provisioner: kubernetes.io/no-provisioner
volumeBindingMode: WaitForFirstConsumer
---
apiVersion: v1
kind: PersistentVolume
metadata:
  name: local-apex19-db
spec:
  capacity:
    storage: 1000Gi
  accessModes:
  - ReadWriteOnce
  persistentVolumeReclaimPolicy: Retain
  storageClassName: local
  local:
    path: /newdisk
  nodeAffinity:
    required:
      nodeSelectorTerms:
      - matchExpressions:
        - key: kubernetes.io/hostname
          operator: In
          values:
          - instance-20191023-0015
---
kind: PersistentVolumeClaim
apiVersion: v1
metadata:
  name: pvc-apex19-db
spec:
  accessModes:
  - ReadWriteOnce
  storageClassName: local
  resources:
    requests:
      storage: 1000Gi
EOF
```

- Disable istio-injection, create a job to copy datafiles from container fs to pv then enable istio-injection
```
kubectl label ns default istio-injection- && \
echo -e 'apiVersion: batch/v1
kind: Job
metadata:
  name: copydb
  annotations:
    sidecar.istio.io/inject: "false"
spec:
  template:
    spec:
      containers:
      - name: apex19
        image: henryhhl18/apex19-container
        env:
        - name: COPYDBFROM
          value: "/u02"
        - name: COPYDBTO
          value: "/pvc_mount_point"
        command: ["/bin/bash","-c","cd $(COPYDBFROM);tar cf - * | ( cd $(COPYDBTO); tar xfp -)"]
        volumeMounts:
        - mountPath: /pvc_mount_point
          name: storage-u02
      volumes:
        - name: storage-u02
          persistentVolumeClaim:
            claimName: pvc-apex19-db
      restartPolicy: Never
  backoffLimit: 1' | tee copydb-job.yaml && \
kubectl apply -f copydb-job.yaml && \
kubectl label ns default istio-injection=enabled
```

- Deployment/Service of Oracle db/apex 19.1
```
cat<<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: apex19
  name: apex19
spec:
  replicas: 1
  selector:
    matchLabels:
      app: apex19
  template:
    metadata:
      labels:
        app: apex19
    spec:
      hostname: dbserv1
      containers:
      - image: henryhhl18/apex19-container
        name: apex19
        ports:
        - containerPort: 1521
        volumeMounts:
        - mountPath: /u02
          name: storage-u02
      volumes:
        - name: storage-u02
          persistentVolumeClaim:
            claimName: pvc-apex19-db
---
apiVersion: v1
kind: Service
metadata:
  name: dbserv1
spec:
  ports:
  - port: 1521
    protocol: TCP
    targetPort: 1521
  selector:
    app: apex19
  type: ClusterIP
EOF
```

- Deployment/Service/Virtualservice of Oracle ords as front-end webapp on Tomcat
```
cat<<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: ords
  name: ords
spec:
  replicas: 2
  selector:
    matchLabels:
      app: ords
  template:
    metadata:
      labels:
        app: ords
    spec:
      hostname: webserv1
      containers:
      - image: henryhhl18/ords-container
        name: ords
        ports:
        - containerPort: 8080
        command: ['/bin/bash', '-c', 'sleep 60s; /bin/bash /tmp/startup-ords']
---
apiVersion: v1
kind: Service
metadata:
  name: ords-service-fe
spec:
  ports:
  - port: 8080
    protocol: TCP
    targetPort: 8080
  selector:
    app: ords
  type: ClusterIP
---
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: ords-vs
spec:
  hosts:
  - 'apex.busyapi.com'
  gateways:
  - istio-system/istio-autogenerated-k8s-ingress
  http:
  - match:
    - uri:
        prefix: /
    route:
    - destination:
        host: ords-service-fe
        port:
          number: 8080
EOF
```

- clean up
```
kubectl delete job.batch/copydb \
service/ords-service-fe \
service/dbserv1 \
deployment/ords \
deployment/apex19 \
vs/ords-vs \
storageclass.storage.k8s.io/local \
persistentvolumeclaim/pvc-apex19-db \
persistentvolume/local-apex19-db
```

- Deployment dependencies control
```
#!/bin/bash
kubectl apply -f APEX-PROD.yaml
while [[ $(kubectl get pods -l run=apex19 -o 'jsonpath={..status.conditions[?(@.type=="Ready")].status}') != "True" ]]; do echo "waiting for pod" && sleep 1; done
kubectl apply -f ORDS-PROD.yaml
```
